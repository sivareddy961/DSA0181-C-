#include <iostream>
using namespace std;

int main() {
    float units, bill = 0.0f;

    cout << "Enter units consumed: ";
    cin >> units;

    if (units <= 100) {
        bill = units * 5;
    } else if (units <= 300) {
        bill = (100 * 5) + ((units - 100) * 8);
    } else {
        bill = (100 * 5) + (200 * 8) + ((units - 300) * 10);
    }

    cout.precision(2);
    cout.setf(ios::fixed);
    
    cout << "Electricity Bill: â‚¹" << bill << endl;

    return 0;
}
